 38  * @param {DBClass.Adapter} type Database type; If Yahoo, this will be ignored.

 42 DBClass.create = function(name,size,type,version,comment) {
 43 	if(!name)  throw Error("Parameter:name is required.");
 44 	var c = function() {
 45 		this.name = name||this.name;
 46 		this.size = size||this.size;
 47 		this.type = type||this.type;
 48 		this.version = version||this.vertion;
 49 		this.comment = this.comment;
 50 		this.db = DBClass.detectDB(this);
 51 	};
 52 	c.prototype = DBClass.prototype;
 53 	return c;
 54 }

 56 /**
 57  * @private
 58  * @static
 59  */
 60 DBClass.detectDB = function(ins) {
 61 	var db;
 62 	switch(ins.type) {
 63 	case DBClass.Adapter.HTML5:
 64 		db = openDatabase(ins.name, ins.version, ins.comment, ins.size);
 65 		return db;
 66 		break;
 67 	case DBClass.Adapter.YAHOO:
 68 		db = new SQLite();
 69 		db.open(ins.name);
 70 		return db;
 71 		break;
 72 	case DBClass.Adapter.AUTO:
 73 		var e;
 74 		for(var i in DBClass.Adapter) {
 75 			var t = DBClass.Adapter[i];
 76 			if(t==DBClass.Adapter.AUTO) continue;
 77 			try {
 78 				ins.type = t;
 79 				db = DBClass.detectDB(ins);
 80 				return db;
 81 			} catch(e) {
 82 				throw Error("Database is not supported.");
 83 			}
 84 		}
 85 		break;
 86 	}
 87 }

151 /** @class */
152 DBClass.Schema.prototype = {
153 	/**
154 	 * Initialize Schema
155 	 * @param {Function} callback
156 	 */
157 	init : function(callback) {
158 		var sql = new AbstractSQL(this.name);
159 		var schema = this;
160 		callback = callback || function(){};
161 		if(!this.createManually) {
162 			schema.exec(sql.createTable(this.fields,true,false),function(){
163 				callback.apply(schema,[this]);
164 			});
165 		}
166 	},
167 	/**
168 	 * If true, don't create table automatic.
169 	 * @type Boolean
170 	 */
171 	createManually : false,
172 	/**
173 	 * Table name
174 	 * @type String
175 	 */
176 	name : "",
177 	/**
178 	 * Array contains AbstractSQL.Field class instances.
179 	 * @type Array.<AbstractSQL.Field>
180 	 */
181 	fields : null,
182 	/**
183 	 * Execute SQL
184 	 * @param {AbstractSQL} sql First callback argument in DBClass.Result instance.
185 	 * @param {Function} callback
186 	 * @param {Array.<String>} bind Bind Variables; Not supported in Yahoo
187 	 */
188 	exec : function(sql,callback,bind) {
189 		var schema = this;
190 		callback = callback || function(){};
191 		bind = bind||[];
192 		if(schema.db.type==DBClass.Adapter.YAHOO) {
193 			
194 		} else {
195 			schema.db.db.transaction(function(tx){
196 				tx.executeSql(sql,bind,function(tx,r){
197 					var res = new DBClass.Result(r.rows&& new RegExp("^select .+","i").test(sql)?r:null);
198 					res.success = true;
199 					res.sql = sql;
200 					callback.apply(res);
201 				},function(tx,err){
202 					var res = new DBClass.Result(null);
203 					res.success = true;
204 					res.sql = sql;
205 					res.error = err;
206 					callback.apply(res);
207 				});
208 			});
209 		}
210 	},
211 	/**
212 	 * Select rows
213 	 * @param {Array.<String>} fields
214 	 * @param {AbstractSQL.Where|AbstractSQL.WhereList} where
215 	 * @param {int} limit
216 	 * @param {AbstractSQL.Order} order
217 	 * @param {Function} callback
218 	 */
219 	select : function(fields,where,limit,order,callback) {
220 		var sql = new AbstractSQL(this.name);
221 		var schema = this;
222 		callback = callback||function(){};
223 		schema.exec(sql.select(fields,where,limit,order),function(){
224 			callback.apply(this,[this.success]);
225 		});
226 		
227 	},
228 	/**
229 	 * Insert data
230 	 * @param {Map.<String,mixed>} data
231 	 * @param {AbstractSQL.Conflict} onConflict
232 	 * @param {Function} callback
233 	 */
234 	insert : function(data,onConflict,callback) {
235 		var sql = new AbstractSQL(this.name);
236 		var schema = this;
237 		callback = callback||function(){};
238 		schema.exec(sql.insert(data,onConflict),function(){
239 			callback.apply(this,[this.success]);
240 		});
241 		
242 	},
243 	/**

247 	 * @param {AbstractSQL.Conflict} on conflict
248 	 * @param {Function} callback

250 	update : function(data,where,onConflict,callback) {
251 		var sql = new AbstractSQL(this.name);
252 		var schema = this;
253 		callback = callback||function(){};

255 			callback.apply(this,[this.success]);
256 		});
257 	},
258 	/**
259 	 * Remove rows
260 	 * @param {AbstractSQL.Where|AbstractSQL.WhereList} where
261 	 * @param {Function} callback
262 	 */
263 	remove : function(where,callback) {
264 		var sql = new AbstractSQL(this.name);
265 		var schema = this;
266 		callback = callback||function(){};
267 		schema.exec(sql.remove(where),function(){
268 			callback.apply(this,[this.success]);
269 		});
270 	},
271 	/**
272 	 * Get rows size, first argument of callback is number.
273 	 * @param {AbstractSQL.Where} where
274 	 * @param {Function} callback
275 	 */
276 	count : function(where,callback) {
277 		var sql = new AbstractSQL(this.name);
278 		var schema = this;
279 		if(typeof where == "function") {
280 			callback = where;
281 			where = null;
282 		}
283 		callback = callback||function(){};
284 		schema.exec(sql.count("*",where),function(){
285 			var r = !this.error&&this.resultSet?this.resultSet.rows:false;
286 			if(!r) return callback.apply(this,[-1]);
287 			var o = r.item(0);
288 			for(var i in o) {
289 				return callback.apply(this,[o[i]]);
290 			}
291 		});
292 	},
293 	/**
294 	 * Drop table
295 	 * @param {Function} callback
296 	 */
297 	drop : function(callback) {
298 		var sql = new AbstractSQL(this.name);
299 		var schema = this;
300 		callback = callback||function(){};
301 		schema.exec(sql.dropTable(true),function(){
302 			callback.apply(this,[!this.error]);
303 		});
304 		
305 	}

308 /**
309  * Called from only DBClass#select
310  * @constructor
311  * @param {SQLResultSet} resultSet
312  */
313 DBClass.Result = function(resultSet) {
314 	if(resultSet) {
315 		this.resultSet = resultSet;
316 		var rows = resultSet.rows;
317 		/** @ignore */
318 		this.each = function(callback) {
319 			for(var i=0;i<rows.length;i++) {
320 				callback.apply(rows.item(i),[i,this]);
321 			}
322 		}
323 		/** @ignore */
324 		this.length = rows.length;
325 		/** @ignore */
326 		this.item = function(idx) {
327 			return rows.item(idx);
328 		}
329 	}
330 }
