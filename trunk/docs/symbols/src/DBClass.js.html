 36 /**
 37  * @static
 38  * @function
 39  * @param {String} name Database name or Path to database(Yahoo).
 40  * @param {int} type Database size; If Yahoo,this will be ignored.
 41  * @param {DBClass.Adapter} type Database type; If Yahoo, this will be ignored.
 42  * @param {String} version Database version; If Yahoo, this will be ignored.
 43  * @param {String} comment Database comment; If Yahoo, this will be ignored.
 44  */
 45 DBClass.create = function(name,size,type,version,comment) {
 46 	if(!name)  throw Error("Parameter:name is required.");
 47 	var c = function() {
 48 		this.name = name||this.name;
 49 		this.size = size||this.size;
 50 		this.type = type||this.type;
 51 		this.version = version||this.vertion;
 52 		this.comment = this.comment;
 53 		this.db = DBClass.detectDB(this);
 54 	};
 55 	c.prototype = DBClass.prototype;
 56 	return c;
 57 }

 59 /**
 60  * @private
 61  * @static
 62  */
 63 DBClass.detectDB = function(ins) {
 64 	var db;
 65 	switch(ins.type) {
 66 	case DBClass.Adapter.HTML5:
 67 		db = openDatabase(ins.name, ins.version, ins.comment, ins.size);
 68 		return db;
 69 		break;
 70 	case DBClass.Adapter.GEARS:
 71 		db = google.gears.factory.create("beta.database");
 72 		db.open(ins.name);
 73 		return db;
 74 		break;
 75 	case DBClass.Adapter.YAHOO:
 76 		db = new SQLite();
 77 		db.open(ins.name);
 78 		return db;
 79 		break;
 80 	case DBClass.Adapter.AIR:
 81 		var f = new air.File("app-storage:/"+ins.name);
 82 		db = {
 83 			file : f,
 84 			connection : new air.SQLConnection()
 85 		};
 86 		return db;
 87 		break;
 88 	case DBClass.Adapter.AUTO:
 89 		var e;
 90 		for(var i in DBClass.Adapter) {
 91 			var t = DBClass.Adapter[i];
 92 			if(t==DBClass.Adapter.AUTO) continue;
 93 			try {
 94 				ins.type = t;
 95 				db = DBClass.detectDB(ins);
 96 				if(db) return db;
 97 			} catch(e) {}
 98 		}
 99 		throw Error("Database is not supported.");
100 		break;
101 	}
102 }

166 /** @class */
167 DBClass.Schema.prototype = {
168 	/**
169 	 * Initialize Schema
170 	 * @param {Function} callback
171 	 */
172 	init : function(callback) {
173 		var sql = new AbstractSQL(this.name);
174 		var schema = this;
175 		var db = schema.db.db;
176 		callback = callback || function(){};
177 		var f = function() {
178 			if(!schema.createManually) {
179 				schema.exec(sql.createTable(schema.fields,true,false),function(){
180 					callback.apply(schema,[this]);
181 				});
182 			}
183 		}
184 		if(schema.db.type==DBClass.Adapter.AIR) {
185 			db.connection.addEventListener(air.SQLEvent.OPEN, function(e){
186 				f();
187 			});
188 			db.connection.addEventListener(air.SQLErrorEvent.ERROR, function(e){
189 				var res = new DBClass.Result(null);
190 				res.success = false;
191 				res.error = e.error;
192 				callback.apply(schema,[res]);
193 			});
194 			db.connection.openAsync(db.file);
195 		} else {
196 			f();
197 		}
198 	},
199 	/**
200 	 * If true, don't create table automatic.
201 	 * @type Boolean
202 	 */
203 	createManually : false,
204 	/**
205 	 * Table name
206 	 * @type String
207 	 */
208 	name : "",
209 	/**
210 	 * Array contains AbstractSQL.Field class instances.
211 	 * @type Array.<AbstractSQL.Field>
212 	 */
213 	fields : null,
214 	/**
215 	 * Execute SQL
216 	 * @param {AbstractSQL} sql First callback argument in DBClass.Result instance.
217 	 * @param {Function} callback
218 	 * @param {Array.<String>} bind Bind Variables; Not supported in Yahoo
219 	 */
220 	exec : function(sql,callback,bind) {
221 		var schema = this;
222 		var isSelect = new RegExp("^select .+","i").test(sql);
223 		var db = schema.db.db;
224 		callback = callback || function(){};
225 		bind = bind||[];
226 		var callbackError = function(err) {
227 			var res = new DBClass.Result(null);
228 			res.success = false;
229 			res.sql = sql;
230 			res.error = err;
231 			callback.apply(res);
232 		}
233 		var callbackSuccess = function(each,item,length) {
234 			var res = new DBClass.Result(isSelect?r:null);
235 			if(isSelect) {
236 				res.each = each;
237 				res.item = item;
238 				res.length = length;
239 			}
240 			res.success = true;
241 			res.sql = sql;
242 			callback.apply(res);
243 		}
244 		switch(schema.db.type) {
245 		case DBClass.Adapter.YAHOO:
246 			try {
247 				var r = isSelect?db.query(sql):db.exec(sql);
248 				if(!r) return callbackError(new Error("No result."));
249 				if(isSelect) {
250 					var rows = r.getAll();
251 					callbackSuccess(
252 						function(cb) {
253 							for(var i=0;i<rows.length;i++) {
254 								cb.apply(this.item(i),[i,this]);
255 							}
256 						},
257 						function(idx) { return rows[idx]; },
258 						rows.length
259 					);
260 					return;
261 				}
262 				callbackSuccess();
263 				return;
264 			} catch(err) {
265 				callbackError(err);
266 				return;
267 			}
268 			callbackError(err);
269 			return;
270 		case DBClass.Adapter.AIR:
271 			if(!db.connection.connected) return callbackError(new Error("Not connected."));
272 			var stmt = new air.SQLStatement();
273 			stmt.sqlConnection = db.connection;
274 			stmt.addEventListener(air.SQLEvent.RESULT, function(e){
275 				
276 			});
277 			stmt.addEventListener(air.SQLErrorEvent.ERROR, function(e){
278 				callbackError(e.error);
279 			});
280 			stmt.text = sql.toString();
281 			stmt.execute();
282 			return;
283 		case DBClass.Adapter.GEARS:
284 			try {
285 				var r = db.execute(sql);
286 				if(isSelect) {
287 					var ar = [];
288 					while(r.isValidRow()){
289 						var itm = {};
290 						for(var i=0;i<r.fieldCount();i++) {
291 							itm[r.fieldName(i)] = r.field(i);
292 						}
293 						ar.push(itm);
294 						r.next();
295 					}
296 					r.close();
297 					callbackSuccess(
298 						function(cb) {
299 							for(var i=0;i<ar.length;i++) {
300 								cb.apply(this.item(i),[i,this]);
301 							}
302 						},
303 						function(idx) { return ar[idx]; },
304 						ar.length
305 					);
306 					return;
307 					
308 				}
309 				if(r) {
310 					r.close();
311 					callbackSuccess();
312 					return;
313 				}
314 			} catch(e) {
315 				callbackError(e);
316 				return;
317 			}
318 			callbackError();
319 			return;
320 		default:
321 			db.transaction(function(tx){
322 				tx.executeSql(sql,bind,function(tx,r){
323 					if(isSelect) {
324 						var rows = r.rows;
325 						callbackSuccess(
326 							function(cb) {
327 								for(var i=0;i<rows.length;i++) {
328 									cb.apply(rows.item(i),[i,this]);
329 								}
330 							},
331 							function(idx) { return rows.item(idx); },
332 							rows.length
333 						);
334 						return;
335 					}
336 					callbackSuccess();
337 					return;
338 				},function(tx,err){
339 					callbackError(err);
340 					return;
341 				});
342 			});
343 			return;
344 		}
345 	},
346 	/**
347 	 * Select rows
348 	 * @param {Array.<String>} fields
349 	 * @param {AbstractSQL.Where|AbstractSQL.WhereList} where
350 	 * @param {int} limit
351 	 * @param {AbstractSQL.Order} order
352 	 * @param {Function} callback
353 	 */
354 	select : function(fields,where,limit,order,callback) {
355 		var sql = new AbstractSQL(this.name);
356 		var schema = this;
357 		callback = callback||function(){};
358 		schema.exec(sql.select(fields,where,limit,order),function(){
359 			callback.apply(this,[this.success]);
360 		});
361 		
362 	},
363 	/**
364 	 * Insert data
365 	 * @param {Map.<String,mixed>} data
366 	 * @param {AbstractSQL.Conflict} onConflict
367 	 * @param {Function} callback
368 	 */
369 	insert : function(data,onConflict,callback) {
370 		var sql = new AbstractSQL(this.name);
371 		var schema = this;
372 		callback = callback||function(){};
373 		schema.exec(sql.insert(data,onConflict),function(){
374 			callback.apply(this,[this.success]);
375 		});
376 		
377 	},
378 	/**
379 	 * Update rows
380 	 * @param {Map.<String,String>} data
381 	 * @param {AbstractSQL.Where|AbstractSQL.WhereList} where
382 	 * @param {AbstractSQL.Conflict} on conflict
383 	 * @param {Function} callback
384 	 */
385 	update : function(data,where,onConflict,callback) {
386 		var sql = new AbstractSQL(this.name);
387 		var schema = this;
388 		callback = callback||function(){};
389 		schema.exec(sql.update(data,where,onConflict),function(){
390 			callback.apply(this,[this.success]);
391 		});
392 	},
393 	/**
394 	 * Remove rows
395 	 * @param {AbstractSQL.Where|AbstractSQL.WhereList} where
396 	 * @param {Function} callback
397 	 */
398 	remove : function(where,callback) {
399 		var sql = new AbstractSQL(this.name);
400 		var schema = this;
401 		callback = callback||function(){};
402 		schema.exec(sql.remove(where),function(){
403 			callback.apply(this,[this.success]);
404 		});
405 	},
406 	/**
407 	 * Get rows size, first argument of callback is number.
408 	 * @param {AbstractSQL.Where} where
409 	 * @param {Function} callback
410 	 */
411 	count : function(where,callback) {
412 		var sql = new AbstractSQL(this.name);
413 		var schema = this;
414 		if(typeof where == "function") {
415 			callback = where;
416 			where = null;
417 		}
418 		callback = callback||function(){};
419 		schema.exec(sql.count("*",where),function(){
420 			var r = !this.error&&this.resultSet?this.resultSet.rows:false;
421 			if(!r) return callback.apply(this,[-1]);
422 			var o = r.item(0);
423 			for(var i in o) {
424 				return callback.apply(this,[o[i]]);
425 			}
426 		});
427 	},
428 	/**
429 	 * Drop table
430 	 * @param {Function} callback
431 	 */
432 	drop : function(callback) {
433 		var sql = new AbstractSQL(this.name);
434 		var schema = this;
435 		callback = callback||function(){};
436 		schema.exec(sql.dropTable(true),function(){
437 			callback.apply(this,[!this.error]);
438 		});
439 		
440 	}
441 }
